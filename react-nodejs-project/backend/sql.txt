CREATE TABLE products (
    id INT PRIMARY KEY AUTO_INCREMENT,
    image_data VARCHAR(500),
    name VARCHAR(255),
    price DECIMAL(10,2),
    stockCode VARCHAR(255),
    stockQuantity INT,
    discountRate DECIMAL(5,2),
    trademark_id INT,
    category_id INT,
    subcategory_id INT,
    additionalfeatures_id INT,
    variants_id INT,
    description TEXT
);
CREATE TABLE additionalfeatures (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL
);
CREATE TABLE variants (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL
);
CREATE TABLE trademarks (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL,
    image_data VARCHAR(500)
);
CREATE TABLE categorys (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL
);
CREATE TABLE subcategorys (
    id INT PRIMARY KEY AUTO_INCREMENT,
    category_id INT,
    name VARCHAR(255)
);
CREATE TABLE product_additionalfeatures (
    id INT PRIMARY KEY AUTO_INCREMENT,
    product_id INT,
    additionalfeature_id INT,
    PRIMARY KEY (product_id, additionalfeature_id)
);
CREATE TABLE product_additionalfeatures_details (
    id INT PRIMARY KEY AUTO_INCREMENT,
    product_id INT,
    additionalfeature_id INT,
    details VARCHAR(255)
);
CREATE TABLE product_variants (
    id INT PRIMARY KEY AUTO_INCREMENT,
    product_id INT,
    variant_id INT,
    PRIMARY KEY (product_id, variant_id)
);
CREATE TABLE product_variant_details (
    id INT PRIMARY KEY AUTO_INCREMENT,
    product_id INT,
    variant_id INT,
    details VARCHAR(255)
);
CREATE TABLE orders (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT,
    order_number BIGINT,
    order_date DATETIME,
    delivery_date DATETIME,
    status VARCHAR(255),
    items JSON
);
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    firstName VARCHAR(255),
    lastName VARCHAR(255),
    email VARCHAR(255),
    password VARCHAR(255),
    phone VARCHAR(20),
    nationalId VARCHAR(20),
    gender VARCHAR(10),
    createdAt DATETIME,
    updatedAt DATETIME
);
